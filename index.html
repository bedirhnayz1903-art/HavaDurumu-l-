import pygame import random # --- OYUNUN AYARLARI --- GENISLIK = 480 YUKSEKLIK = 600 FPS = 60 # Saniyede 60 kare (Akıcı oyun için) # Renkler SIYAH = (0, 0, 0) BEYAZ = (255, 255, 255) KIRMIZI = (255, 0, 0) SARI = (255, 255, 0) # --- OYUNCU (UÇAK) SINIFI --- class Oyuncu(pygame.sprite.Sprite): def __init__(self): super().__init__() self.image = pygame.Surface((50, 40)) # Uçağın boyutu self.image.fill((0, 255, 0)) # Gerçek oyunda buraya uçağın resmi gelir self.rect = self.image.get_rect() self.rect.centerx = GENISLIK / 2 self.rect.bottom = YUKSEKLIK - 10 self.hiz_x = 0 def update(self): # Hareket kontrolü self.hiz_x = 0 tuslar = pygame.key.get_pressed() if tuslar[pygame.K_LEFT]: self.hiz_x = -8 if tuslar[pygame.K_RIGHT]: self.hiz_x = 8 self.rect.x += self.hiz_x def ates_et(self): mermi = Mermi(self.rect.centerx, self.rect.top) tum_spritler.add(mermi) mermiler.add(mermi) # --- MERMİ SINIFI --- class Mermi(pygame.sprite.Sprite): def __init__(self, x, y): super().__init__() self.image = pygame.Surface((10, 20)) self.image.fill(SARI) self.rect = self.image.get_rect() self.rect.bottom = y self.rect.centerx = x self.hiz_y = -10 # Mermi yukarı doğru gider def update(self): self.rect.y += self.hiz_y # Ekrandan çıkarsa mermiyi sil (Performans için) if self.rect.bottom < 0: self.kill() # --- DÜŞMAN SINIFI --- class Dusman(pygame.sprite.Sprite): def __init__(self): super().__init__() self.image = pygame.Surface((30, 30)) self.image.fill(KIRMIZI) self.rect = self.image.get_rect() self.rect.x = random.randrange(GENISLIK - self.rect.width) self.rect.y = random.randrange(-100, -40) self.hiz_y = random.randrange(1, 8) def update(self): self.rect.y += self.hiz_y if self.rect.top > YUKSEKLIK + 10: self.rect.x = random.randrange(GENISLIK - self.rect.width) self.rect.y = random.randrange(-100, -40) self.hiz_y = random.randrange(1, 8) # --- OYUNU BAŞLATMA --- pygame.init() ekran = pygame.display.set_mode((GENISLIK, YUKSEKLIK)) pygame.display.set_caption("1945 Pro Uçak Savaşı") saat = pygame.time.Clock() tum_spritler = pygame.sprite.Group() dusmanlar = pygame.sprite.Group() mermiler = pygame.sprite.Group() oyuncu = Oyuncu() tum_spritler.add(oyuncu) for i in range(8): d = Dusman() tum_spritler.add(d) dusmanlar.add(d) # --- ANA OYUN DÖNGÜSÜ --- oyun_calisiyor = True while oyun_calisiyor: saat.tick(FPS) for event in pygame.event.get(): if event.type == pygame.QUIT: oyun_calisiyor = False elif event.type == pygame.KEYDOWN: if event.key == pygame.K_SPACE: oyuncu.ates_et() # Güncelleme tum_spritler.update() # Mermi düşmana değdi mi kontrolü? vuruslar = pygame.sprite.groupcollide(dusmanlar, mermiler, True, True) for vurus in vuruslar: yeni_dusman = Dusman() tum_spritler.add(yeni_dusman) dusmanlar.add(yeni_dusman) # Düşman oyuncuya çarptı mı? carpisma = pygame.sprite.spritecollide(oyuncu, dusmanlar, False) if carpisma: oyun_calisiyor = False # Game Over! # Çizim ekran.fill(SIYAH) tum_spritler.draw(ekran) pygame.display.flip() pygame.quit()
